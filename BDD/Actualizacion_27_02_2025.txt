USE Software;

DROP TABLE IF EXISTS MUEBLES;
DROP TABLE IF EXISTS Clientes;
DROP TABLE IF EXISTS Usuarios;

CREATE TABLE Usuarios (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    apellido VARCHAR(100),
    nombre_usuario VARCHAR(100),
    contraseña_usuario VARCHAR(50)    
);

CREATE TABLE Clientes (
    ID_Cliente INT AUTO_INCREMENT PRIMARY KEY,
    nombre_cliente VARCHAR(100),
    direccion_cliente VARCHAR(225),
    numero_telf VARCHAR(15),
    RFC VARCHAR(20),
    CURP VARCHAR(20)
);

CREATE TABLE MUEBLES (
    ID_muebles INT AUTO_INCREMENT PRIMARY KEY,
    nombre_mueble VARCHAR(100),
    direccion VARCHAR(225),
    monto_total DECIMAL(10, 2),
    estatus_producto VARCHAR(50),
    id_cliente INT,  -- Relación con la tabla Clientes
    CONSTRAINT fk_cliente FOREIGN KEY (id_cliente) REFERENCES Clientes(ID_Cliente) -- Clave foránea a Clientes
);

CREATE TABLE ADMINISTRADOR (
    NombreAdmin VARCHAR(50) NOT NULL,
    Contraseña VARCHAR(255) NOT NULL
);
